Example
import _ from 'lodash'
import Serp from 'vbda/components/serp'
import CustomList from './vbda/plugins/list'

React.createClass({
    showInfo(hit) {
        // hit == { _index:'friends-yyyy-mm-dd', type:'logs', _id:'xxxxx', _source:{} }
    },
    render() {
        return <Serp
            url='/vbda/search'
            lng='en_us'
            dts={['friends']}
            query={{'line':'line id', phone:'0922'}}
            cfg={{
                data_sources:{
                    osint:{
                        name: 'osint',
                        data_types:['friends']
                    }
                }, 
                data_types:{
                    friends:{
                        renderSerp:['t1','t2']
                    }
                },
                renders:{
                    t1:{   
                        type:'table'
                    },
                    t2:{
                        type: 'custom',
                        vis: CustomList
                    }
                }
            }}
            onSelect={this.showInfo}
            onMouseOver={this.showInfo}
            onClick={this.showInfo} />
    }
})
