const Form = React.createClass({
    propTypes: {
        id: React.PropTypes.string,
        className: React.PropTypes.string,
        lng: React.PropTypes.string,
        cfg: React.PropTypes.shape({
            name: React.PropTypes.string,
            fields: React.PropTypes.objectOf(React.PropTypes.shape({
                title: React.PropTypes.string,
                type: React.PropTypes.oneOf([
                    'string','textarea','dropdown','checkbox','checkbox-group',
                    'date','date-range','radio-group','gis'
                ]),
                list: React.PropTypes.objectOf(React.PropTypes.string)
            })).isRequired,
            locales: React.PropTypes.objectOf(React.PropTypes.shape({
                fields: React.PropTypes.shape({
                    title: React.PropTypes.string,
                    list: React.PropTypes.objectOf(React.PropTypes.string)
                })
            }))
        }).isRequired,
        onSearch: React.PropTypes.func.isRequired
    },
    getDefaultProps() {
        return {
        }
    },
    getInitialState() {
        return {

        }
    },
    handleSearch() {
        const {onSearch} = this.props
        onSearch(this.state)
    },
    render() {
        const {id, className, cfg:{fields}} = this.props
        return <div id={id} className={cx('c-box seamless c-vbda-vis-form', className)}>
            <div className='content'>
            {JSON.stringify(fields, null, 4)}
            </div>
            <footer>
                <button className='pure-button' onClick={this.handleSearch}>Search</button>
            </footer>
        </div>
    }
})

export default localize(Form, 'cfg')
