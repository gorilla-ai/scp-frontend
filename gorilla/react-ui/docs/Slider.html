Example
// controlled

import Slider from 'core/components/slider'
React.createClass({
    getInitialState() {
        return {value:40}
    },
    handleChange(e) {
        let value = e
        this.setState({value})
    },
    render() {
        let {value} = this.state
        return <Slider value={value} onChange={this.handleChange} showProgress={true} min={0} max={100} step={5} />
    }
})
